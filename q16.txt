#include <stdio.h>
#include <string.h>

int main(void) {
  char texto[100];
  int i, tam;
  puts("Digite uma frase:");
  fgets(texto, 100, stdin);

  tam = strlen(texto);
  for(i=tam-1; i>=0; i--){
    printf("%c", texto[i]);
  }
  
  return 0;
}
