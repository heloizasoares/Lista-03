#include <stdio.h>

int main(void) {
  int i, num, contador = 0;

  for(i = 1; i <= 10; i++){
    printf("Digite um numero:");
    scanf("%d", &num);

    if((num >= 10) && (num<=50)){
      contador++;
    }
  }
  printf("%d numeros estao no intervalo\n", contador);
  printf("%d numeros não estao no intervalo", 10-contador);

  return 0;
}
