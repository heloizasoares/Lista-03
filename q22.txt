  for (i=1; i<=tam; i++){

  }
  for(i=1; i<=tam; i++){
    if(A[i] > m){
      m = A[i];
    }
  }
  printf("O seu conjunto é: {");

  for (i=1; i<=tam; i++){
    if (i == tam){
      printf ("%d", A[i]);
    }

      printf ("%d, ", A[i]);
    }
  }
  printf ("} \n");
  a = 1;

  while (a == 1){  
    i = 1;  
    while(i <= tam){
      if (A[i] >= m){
        printf ("*");
      }

          printf(" ");
      }
      i++;      
  }
    printf ("\n");
    m = m-1;

    }
  }
}
