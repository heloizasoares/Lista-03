#include <stdio.h>
#include <string.h>

int main(void) {
  
  char texto[100], i, contador = 0;

  puts("Digite uma frase(max de 100 caract): ");
  fgets(texto, 100, stdin);

  for(i = 0; i<100;i++){
    if(texto[i] == 65 || texto[i] == 97){
      contador = contador + 1;
      
    }
  }
  puts(texto);
  printf("%d", contador);
  return 0;
}
