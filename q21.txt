#include <stdio.h>
#include <stdlib.h>

void funcao(){
  int i, j, tam;

  printf("Digite o tamanho do array: ");
  scanf("%d", &tam);
  int a[tam];

  printf("Preencha seu array: \n");
  for (i=1; i<=tam; i++){
    scanf ("%d", &a[i]);
  }

  printf("O seu conjunto é: {");
  for (i=1; i<=tam; i++){
    if (i==tam){
      printf ("%d", a[i]);
    }
    else{
        printf ("%d, ", a[i]);
    }    
  }
  printf ("} \n");
  for (i=1; i<=tam; i++){
    for (j=1; j<=a[i]; j++){
      printf ("*");
    }
    printf ("\n");
  }
}

int main(void) {
  funcao();

  return 0;
}
