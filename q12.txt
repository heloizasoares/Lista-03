#include <stdio.h>

int main(void) {
  int num, i, contador = 0;

  printf("Digite um numero:");
  scanf("%d", &num);

  for(i = 2; i<10;i++){
    if(num%i == 0){
      contador++;
    }
  }
  if(contador >= 2){
    printf("Nao é primo");
  }
  else{
    printf("Numero primo");
  }
  return 0;
}
